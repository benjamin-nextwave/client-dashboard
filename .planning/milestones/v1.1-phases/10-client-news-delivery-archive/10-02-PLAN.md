```typescript
'use server'

import { createClient } from '@/lib/supabase/server'

/**
 * Dismiss a news item for the currently-authenticated user.
 *
 * Security notes (per Phase 10 threat model):
 * - user_id is read SERVER-SIDE via supabase.auth.getUser() — NEVER accepted from
 *   the caller. This defends against T-1 (cross-user dismissal forgery): a malicious
 *   client cannot pass another user's id even via DevTools.
 * - This uses the REQUEST-SCOPED supabase client (createClient), NOT the admin
 *   client. RLS WITH CHECK (user_id = auth.uid()) on news_dismissals is the
 *   backstop — even if the auth.uid() read above were bypassed, RLS rejects
 *   any INSERT where user_id != auth.uid().
 * - upsert + ignoreDuplicates makes the action idempotent on double-click. The
 *   composite PK (user_id, news_item_id) defines the conflict target.
 * - We deliberately do NOT call revalidatePath. Queue state in the overlay is
 *   client-managed during the session; the next /dashboard render naturally
 *   re-queries `news_items LEFT JOIN news_dismissals` and the dismissed item
 *   is excluded.
 */
export async function dismissNewsItem(
  newsItemId: string
): Promise<{ ok: true } | { error: string }> {
  // Defensive input check — RLS would reject malformed UUIDs anyway, but a clean
  // early return gives a deterministic error message.
  if (typeof newsItemId !== 'string' || newsItemId.length === 0) {
    return { error: 'invalid id' }
  }

  const supabase = await createClient()

  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return { error: 'unauthenticated' }

  const { error } = await supabase
    .from('news_dismissals')
    .upsert(
      { user_id: user.id, news_item_id: newsItemId },
      { onConflict: 'user_id,news_item_id', ignoreDuplicates: true }
    )

  if (error) return { error: error.message }
  return { ok: true }
}
```

### Critical rules

- File starts with `'use server'` (REQUIRED — this is a server action).
- Import `createClient` from `@/lib/supabase/server` — NOT `@/lib/supabase/admin`. The request-scoped client carries the user's session cookies and enforces RLS (D-16).
- DO NOT add any other exports. This file holds exactly one server action; future client-side actions can be added later.
- DO NOT export types — server-action files in this codebase export only async functions (matches the Phase 9 lesson 040a936: re-exporting types from `'use server'` files breaks the client bundle).
- DO NOT add a `prevState` parameter. This action is called via `useTransition` (not `useActionState`), so the signature is just `(newsItemId: string)`. Matches the Phase 9 publish/withdraw pattern.
- DO NOT call `revalidatePath` (D-16) — queue state is client-managed; next page render naturally re-queries.
- The `onConflict: 'user_id,news_item_id'` string MUST exactly match the composite-PK columns (no spaces). The upsert ignores duplicates via `ignoreDuplicates: true`.
- Do NOT pass `dismissed_at` — the column has `DEFAULT NOW()` server-side and the action shouldn't override it.
- The function is the only export. No utility helpers needed.
